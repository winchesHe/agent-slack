```ts
import { consola } from 'consola'
import { createApplication } from '@/application/createApplication.ts'
import { existsSync } from 'node:fs'
import path from 'node:path'

export async function startCommand(opts: { cwd: string }): Promise<void> {
  const configDir = path.join(opts.cwd, '.agent-slack')
  if (!existsSync(configDir)) {
    consola.error(`未找到 ${configDir}`)
    consola.info('请先运行 agent-slack onboard')
    process.exit(1)
  }
  const app = await createApplication({ workspaceDir: opts.cwd })
  await app.start()
  consola.success('agent-slack 已启动，等待 Slack 事件。Ctrl+C 退出。')

  const shutdown = async (signal: string): Promise<void> => {
    consola.info(`收到 ${signal}，正在关闭…`)
    await app.stop()
    app.abortRegistry.abortAll('shutdown')
    process.exit(0)
  }
  process.on('SIGINT', () => { void shutdown('SIGINT') })
  process.on('SIGTERM', () => { void shutdown('SIGTERM') })
}
```

```ts
import { WebClient } from '@slack/web-api'

export type ValidationResult = { ok: true; [k: string]: unknown } | { ok: false; reason: string }

export async function validateLiteLLM(args: {
  baseUrl: string
  apiKey: string
  fetcher?: typeof fetch
}): Promise<ValidationResult> {
  const f = args.fetcher ?? fetch
  try {
    const res = await f(`${args.baseUrl.replace(/\/$/, '')}/health/liveliness`, {
      headers: { Authorization: `Bearer ${args.apiKey}` },
    })
    if (res.status >= 200 && res.status < 300) return { ok: true }
    return { ok: false, reason: `HTTP ${res.status}` }
  } catch (err) {
    return { ok: false, reason: err instanceof Error ? err.message : String(err) }
  }
}

export async function validateSlack(args: {
  botToken: string
  webClientFactory?: (token: string) => Pick<WebClient, 'auth'>
}): Promise<ValidationResult> {
  try {
    const web = (args.webClientFactory ?? ((t) => new WebClient(t)))(args.botToken)
    const r = await web.auth.test()
    if (r.ok) return { ok: true, team: r.team, user: r.user }
    return { ok: false, reason: 'auth.test returned ok=false' }
  } catch (err) {
    return { ok: false, reason: err instanceof Error ? err.message : String(err) }
  }
}
```

```ts
import { consola } from 'consola'
import 'dotenv/config'
import { existsSync } from 'node:fs'
import { resolveWorkspacePaths } from '@/workspace/paths.ts'
import { loadWorkspaceContext } from '@/workspace/WorkspaceContext.ts'
import { validateLiteLLM, validateSlack } from '../validators.ts'
import { createLogger } from '@/logger/logger.ts'
import { createRedactor } from '@/logger/redactor.ts'

export async function doctorCommand(opts: { cwd: string }): Promise<void> {
  let failures = 0
  const check = (label: string, ok: boolean, hint?: string) => {
    consola[ok ? 'success' : 'error'](`${label}: ${ok ? 'OK' : 'FAIL'}${hint ? ` — ${hint}` : ''}`)
    if (!ok) failures++
  }

  // 1. Node
  const nodeMajor = Number(process.versions.node.split('.')[0])
  check('Node >= 22', nodeMajor >= 22, `当前 ${process.versions.node}`)

  // 2. 目录结构
  const paths = resolveWorkspacePaths(opts.cwd)
  check('.agent-slack 存在', existsSync(paths.root))
  check('config.yaml 存在', existsSync(paths.configFile))
  check('system.md 存在', existsSync(paths.systemFile))

  // 3. 凭证
  const slackToken = process.env.SLACK_BOT_TOKEN
  const litellmUrl = process.env.LITELLM_BASE_URL
  const litellmKey = process.env.LITELLM_API_KEY
  check('SLACK_BOT_TOKEN 非空', Boolean(slackToken))
  check('LITELLM_BASE_URL 非空', Boolean(litellmUrl))
  check('LITELLM_API_KEY 非空', Boolean(litellmKey))

  if (slackToken) {
    const r = await validateSlack({ botToken: slackToken })
    check('Slack auth.test', r.ok, r.ok ? undefined : r.reason)
  }
  if (litellmUrl && litellmKey) {
    const r = await validateLiteLLM({ baseUrl: litellmUrl, apiKey: litellmKey })
    check('LiteLLM /health', r.ok, r.ok ? undefined : r.reason)
  }

  // 4. 模型可达（若 litellm 可达）
  if (litellmUrl && litellmKey) {
    try {
      const ctx = await loadWorkspaceContext(opts.cwd, createLogger({
        level: 'warn', redactor: createRedactor([litellmKey]),
      }))
      const res = await fetch(`${litellmUrl.replace(/\/$/, '')}/models`, {
        headers: { Authorization: `Bearer ${litellmKey}` },
      })
      const data = (await res.json()) as { data?: Array<{ id?: string }> }
      const available = (data.data ?? []).map((m) => m.id)
      check(`模型 ${ctx.config.agent.model} 可用`, available.includes(ctx.config.agent.model),
        `可用模型: ${available.slice(0, 5).join(', ')}${available.length > 5 ? '…' : ''}`)
    } catch (err) {
      check('获取模型列表', false, err instanceof Error ? err.message : String(err))
    }
  }

  // 5. Skills 加载
  try {
    const ctx = await loadWorkspaceContext(opts.cwd, createLogger({
      level: 'warn', redactor: createRedactor([]),
    }))
    check(`skills 加载 (${ctx.skills.length} 个)`, true)
  } catch (err) {
    check('skills 加载', false, err instanceof Error ? err.message : String(err))
  }

  if (failures > 0) {
    consola.error(`doctor 发现 ${failures} 个问题`)
    process.exit(1)
  }
  consola.success('全部检查通过')
}
```

```ts
import { consola } from 'consola'
import { existsSync } from 'node:fs'
import { readdir, readFile } from 'node:fs/promises'
import path from 'node:path'
import { loadWorkspaceContext } from '@/workspace/WorkspaceContext.ts'
import { createLogger } from '@/logger/logger.ts'
import { createRedactor } from '@/logger/redactor.ts'
import type { SessionMeta } from '@/store/SessionStore.ts'

export async function statusCommand(opts: { cwd: string }): Promise<void> {
  const logger = createLogger({ level: 'warn', redactor: createRedactor([]) })
  const ctx = await loadWorkspaceContext(opts.cwd, logger)

  consola.box([
    `Workspace: ${opts.cwd}`,
    `Agent:     ${ctx.config.agent.name} / ${ctx.config.agent.model}`,
    `Skills:    ${ctx.skills.length}${ctx.skills.length > 0 ? ' — ' + ctx.skills.map((s) => s.name).join(', ') : ''}`,
  ].join('\n'))

  const slackDir = path.join(ctx.paths.sessionsDir, 'slack')
  if (!existsSync(slackDir)) { consola.info('暂无 Slack session'); return }
  const dirs = await readdir(slackDir)
  const metas: Array<SessionMeta & { dir: string }> = []
  for (const d of dirs) {
    const metaFile = path.join(slackDir, d, 'meta.json')
    if (!existsSync(metaFile)) continue
    try {
      const m = JSON.parse(await readFile(metaFile, 'utf8')) as SessionMeta
      metas.push({ ...m, dir: d })
    } catch { /* skip */ }
  }
  metas.sort((a, b) => b.updatedAt.localeCompare(a.updatedAt))
  const top = metas.slice(0, 5)
  consola.log('\n最近 session:')
  for (const m of top) {
    consola.log(`  #${m.channelName} (${m.channelId}) · ${m.updatedAt} · ${m.usage.stepCount} steps · $${m.usage.totalCostUSD.toFixed(4)}`)
  }
}
```

```ts
import * as clack from '@clack/prompts'

export interface Prompter {
  text(opts: { message: string; initialValue?: string; validate?: (v: string) => string | undefined }): Promise<string>
  password(opts: { message: string; validate?: (v: string) => string | undefined }): Promise<string>
  select<T>(opts: { message: string; options: Array<{ label: string; value: T }>; initialValue?: T }): Promise<T>
  confirm(opts: { message: string; initialValue?: boolean }): Promise<boolean>
  intro(msg: string): void
  outro(msg: string): void
  note(msg: string, title?: string): void
  cancel(msg: string): never
}

export function createClackPrompter(): Prompter {
  return {
    async text(opts) {
      const v = await clack.text(opts)
      if (clack.isCancel(v)) throw new Error('cancelled')
      return v
    },
    async password(opts) {
      const v = await clack.password(opts)
      if (clack.isCancel(v)) throw new Error('cancelled')
      return v
    },
    async select<T>(opts: { message: string; options: Array<{ label: string; value: T }>; initialValue?: T }): Promise<T> {
      const v = await clack.select<T>(opts as never)
      if (clack.isCancel(v)) throw new Error('cancelled')
      return v as T
    },
    async confirm(opts) {
      const v = await clack.confirm(opts)
      if (clack.isCancel(v)) throw new Error('cancelled')
      return Boolean(v)
    },
    intro: (m) => clack.intro(m),
    outro: (m) => clack.outro(m),
    note: (m, t) => clack.note(m, t),
    cancel: (m) => { clack.cancel(m); process.exit(1) },
  }
}
```

7. 提示 `agent-slack start`

**核心是把业务逻辑从 clack IO 解耦**：`runOnboard({ cwd, prompter, fs, validators })` 接受注入 → 可单测。

```ts
import { existsSync } from 'node:fs'
import { mkdir, readFile, writeFile, appendFile } from 'node:fs/promises'
import path from 'node:path'
import { resolveWorkspacePaths } from '@/workspace/paths.ts'
import { createClackPrompter, type Prompter } from '../prompts.ts'
import { defaultConfigYaml, defaultSystemMd, defaultEnv } from '../templates.ts'
import { validateSlack, validateLiteLLM } from '../validators.ts'

export interface OnboardDeps {
  cwd: string
  prompter?: Prompter
}

export async function onboardCommand(deps: OnboardDeps): Promise<void> {
  const prompter = deps.prompter ?? createClackPrompter()
  const paths = resolveWorkspacePaths(deps.cwd)
  prompter.intro(`agent-slack onboard — workspace: ${deps.cwd}`)

  if (existsSync(paths.root)) {
    const action = await prompter.select<'fill' | 'overwrite' | 'exit'>({
      message: `${paths.root} 已存在`,
      options: [
        { label: '补齐缺失字段', value: 'fill' },
        { label: '完全覆盖', value: 'overwrite' },
        { label: '退出', value: 'exit' },
      ],
    })
    if (action === 'exit') { prompter.outro('已取消'); return }
    if (action === 'fill') { prompter.note('补齐模式：仅创建缺失的文件，不覆盖已有字段', '提示') }
  }

  const slackBotToken = await prompter.password({
    message: 'SLACK_BOT_TOKEN (xoxb-...)',
    validate: (v) => (v.startsWith('xoxb-') ? undefined : '应以 xoxb- 开头'),
  })
  const slackAppToken = await prompter.password({
    message: 'SLACK_APP_TOKEN (xapp-...)',
    validate: (v) => (v.startsWith('xapp-') ? undefined : '应以 xapp- 开头'),
  })
  const slackSigningSecret = await prompter.password({ message: 'SLACK_SIGNING_SECRET' })
  const litellmBaseUrl = await prompter.text({
    message: 'LiteLLM Base URL',
    initialValue: 'http://localhost:4000',
  })
  const litellmApiKey = await prompter.password({ message: 'LiteLLM API Key' })
  const model = await prompter.text({
    message: '默认模型',
    initialValue: 'claude-sonnet-4-6',
  })

  const slackRes = await validateSlack({ botToken: slackBotToken })
  prompter.note(
    slackRes.ok ? `✓ Slack auth.test 通过（team=${(slackRes as any).team}）` : `✗ Slack: ${slackRes.reason}`,
    'Slack',
  )
  const liteRes = await validateLiteLLM({ baseUrl: litellmBaseUrl, apiKey: litellmApiKey })
  prompter.note(
    liteRes.ok ? '✓ LiteLLM /health 通过' : `✗ LiteLLM: ${liteRes.reason}（可稍后 doctor）`,
    'LiteLLM',
  )

  if (!slackRes.ok || !liteRes.ok) {
    const go = await prompter.confirm({ message: '有校验失败。仍要写入配置？', initialValue: true })
    if (!go) { prompter.outro('已取消'); return }
  }

  await mkdir(paths.sessionsDir, { recursive: true })
  await mkdir(paths.memoryDir, { recursive: true })
  await mkdir(paths.skillsDir, { recursive: true })
  await mkdir(paths.logsDir, { recursive: true })

  if (!existsSync(paths.configFile)) {
    await writeFile(paths.configFile, defaultConfigYaml(model))
  }
  if (!existsSync(paths.systemFile)) {
    await writeFile(paths.systemFile, defaultSystemMd())
  }
  const envFile = path.join(paths.root, '.env.local')
  await writeFile(envFile, defaultEnv({
    slackBotToken, slackAppToken, slackSigningSecret, litellmBaseUrl, litellmApiKey,
  }))

  await maybeAppendGitignore(deps.cwd)

  prompter.outro([
    `✓ 已初始化 ${paths.root}`,
    `下一步：agent-slack start`,
  ].join('\n'))
}

async function maybeAppendGitignore(cwd: string): Promise<void> {
  const gi = path.join(cwd, '.gitignore')
  if (!existsSync(gi)) return
  const content = await readFile(gi, 'utf8')
  if (content.includes('.agent-slack/')) return
  await appendFile(gi, `\n# agent-slack\n.agent-slack/sessions/\n.agent-slack/logs/\n.agent-slack/.env.local\n`)
}
```

> Spec §3.2 明确 `.env.local` 放 `<cwd>/.agent-slack/`（per-workspace）。`.gitignore` 补丁只 ignore `sessions/` / `logs/` / `.env.local`——配置本身（config.yaml / system.md / skills/）应该被 git 跟踪。
