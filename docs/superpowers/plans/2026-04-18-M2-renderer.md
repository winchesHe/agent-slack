```ts
import type { Logger } from '@/logger/logger.ts'

export type SafeRender = <T>(label: string, fn: () => Promise<T>) => Promise<T | undefined>

export function createSafeRender(logger: Logger): SafeRender {
  const log = logger.withTag('slack:render')
  return async (label, fn) => {
    try {
      return await fn()
    } catch (err) {
      log.warn(`slack api failed [${label}]`, err)
      return undefined
    }
  }
}
```

```ts
import type { AgentExecutionEvent } from '@/core/events.ts'
import type { TotalUsage } from '@/core/usage.ts'

export interface ToolCallRecord {
  id: string
  name: string
  phase: 'input' | 'running' | 'done' | 'error'
  inputPartial: string
  input?: unknown
  outputSummary?: string
}

export interface RenderState {
  phase: 'thinking' | 'streaming' | 'done' | 'error'
  thinkingTick: number
  bufferedText: string
  reasoningText: string
  toolCalls: ToolCallRecord[]
  errorMessage?: string
  finalTotalUsage?: TotalUsage
}

export function initialRenderState(): RenderState {
  return {
    phase: 'thinking',
    thinkingTick: 0,
    bufferedText: '',
    reasoningText: '',
    toolCalls: [],
  }
}

export function reduceRenderState(state: RenderState, event: AgentExecutionEvent): RenderState {
  switch (event.type) {
    case 'text_delta':
      return { ...state, phase: 'streaming', bufferedText: state.bufferedText + event.text }
    case 'reasoning_delta':
      return { ...state, reasoningText: state.reasoningText + event.text }
    case 'tool_input_delta': {
      const calls = upsert(state.toolCalls, event.toolCallId, (c) => ({
        ...c,
        name: event.toolName || c.name,
        phase: c.phase === 'done' || c.phase === 'error' ? c.phase : 'input',
        inputPartial: c.inputPartial + event.partial,
      }))
      return { ...state, toolCalls: calls }
    }
    case 'tool_call_start': {
      const calls = upsert(state.toolCalls, event.toolCallId, (c) => ({
        ...c,
        name: event.toolName,
        phase: 'running',
        input: event.input,
      }))
      return { ...state, toolCalls: calls }
    }
    case 'tool_call_end': {
      const calls = state.toolCalls.map((c) =>
        c.id === event.toolCallId
          ? {
              ...c,
              phase: event.isError ? ('error' as const) : ('done' as const),
              outputSummary: summarizeOutput(event.output),
            }
          : c,
      )
      return { ...state, toolCalls: calls }
    }
    case 'step_start':
    case 'step_finish':
      return state
    case 'done':
      return {
        ...state,
        phase: 'done',
        bufferedText: event.finalText || state.bufferedText,
        finalTotalUsage: event.totalUsage,
      }
    case 'error':
      return { ...state, phase: 'error', errorMessage: event.error.message }
  }
}

function upsert(
  list: ToolCallRecord[],
  id: string,
  mut: (c: ToolCallRecord) => ToolCallRecord,
): ToolCallRecord[] {
  const idx = list.findIndex((c) => c.id === id)
  if (idx === -1) {
    return [
      ...list,
      mut({ id, name: '', phase: 'input', inputPartial: '' }),
    ]
  }
  return list.map((c, i) => (i === idx ? mut(c) : c))
}

function summarizeOutput(output: unknown): string {
  if (typeof output === 'string') return truncate(output, 200)
  try { return truncate(JSON.stringify(output), 200) } catch { return '[unserializable]' }
}

function truncate(s: string, n: number): string {
  return s.length > n ? s.slice(0, n) + '…' : s
}
```

```ts
import type { TotalUsage } from '@/core/usage.ts'

export interface SlackContextBlock {
  type: 'context'
  elements: Array<{ type: 'mrkdwn'; text: string }>
}

export function renderCostContext(usage: TotalUsage): SlackContextBlock {
  const parts: string[] = []
  parts.push(`${(usage.durationMs / 1000).toFixed(1)}s`)
  if (usage.totalCostUSD !== undefined) parts.push(`$${usage.totalCostUSD.toFixed(4)}`)
  const tokens = usage.inputTokens + usage.outputTokens
  let tokenDesc = `${usage.model}: ${tokens.toLocaleString()} tokens`
  if (usage.cacheHitRate !== undefined && usage.cachedInputTokens > 0) {
    tokenDesc += ` (${Math.round(usage.cacheHitRate * 100)}% cache hit)`
  }
  parts.push(tokenDesc)
  return {
    type: 'context',
    elements: [{ type: 'mrkdwn', text: parts.join(' · ') }],
  }
}
```

**策略**：

```ts
import { markdownToBlocks } from 'markdown-to-slack-blocks'

const MAX_CHARS_PER_MESSAGE = 38_000

export interface SplitResult {
  main: unknown[]        // Slack Block
  rest: unknown[][]
}

export function splitMarkdown(md: string): SplitResult {
  const all = markdownToBlocks(md)
  const groups: unknown[][] = [[]]
  let current = 0
  for (const block of all) {
    const size = JSON.stringify(block).length
    if (size + current > MAX_CHARS_PER_MESSAGE && groups.at(-1)!.length > 0) {
      groups.push([])
      current = 0
    }
    groups.at(-1)!.push(block)
    current += size
  }
  return { main: groups[0] ?? [], rest: groups.slice(1) }
}
```

**职责**：纯函数 `render(state, { thinkingIdx }) → { text, blocks[] }`。不触发任何副作用。

**渲染规则（按 phase）**：
- `thinking`（还没有文本 / tool）：`blocks = [section: "🤔 {thinkingPhrase}…"]`，`text = "{thinkingPhrase}…"`（text fallback）
- `streaming`：
  - toolCalls 区（context block）：`🔧 bash · ⚙️ edit_file(running)`（简洁单行）
  - reasoning 折叠（若 `reasoningText.length > 0`）：context block `🤔 {reasoningTextLastN}` 截断 500 char
  - 主消息：`bufferedText` markdown
- `done`：
  - toolCalls 最终历史（只显示 name + 成功/失败标记）
  - 主文本 = `bufferedText`（最终版）
  - context block = cost（若有 totalUsage）
- `error`：
  - 主文本 = `⚠️ 错误: {errorMessage}`
  - 保留 toolCalls 历史（如果有）

- [ ] **Step 1: 写测试（每种 phase 一个）**

覆盖 `thinking` / `streaming with tool` / `done with cost` / `error` 四种形态，断言 `blocks` 包含关键关键字（`"thinking"` / `"bash"` / `"cache hit"` / `"⚠️"`）。

```ts
import type { RenderState, ToolCallRecord } from './renderState.ts'
import { pickThinkingPhrase } from './thinkingRotation.ts'
import { renderCostContext } from './costBlock.ts'
import { splitMarkdown } from './messageSplitter.ts'

export interface RenderOutput {
  text: string
  blocks: unknown[]
  restBlocks: unknown[][]     // 需要另发的分段（若长消息）
}

export function renderSlack(state: RenderState): RenderOutput {
  if (state.phase === 'thinking') {
    const phrase = pickThinkingPhrase(state.thinkingTick)
    return {
      text: `${phrase}…`,
      blocks: [
        { type: 'section', text: { type: 'mrkdwn', text: `🤔 ${phrase}…` } },
      ],
      restBlocks: [],
    }
  }

  if (state.phase === 'error') {
    const blocks: unknown[] = [
      {
        type: 'section',
        text: { type: 'mrkdwn', text: `⚠️ 错误: ${state.errorMessage ?? 'unknown'}` },
      },
    ]
    const tool = renderToolHistory(state.toolCalls)
    if (tool) blocks.push(tool)
    return { text: `⚠️ ${state.errorMessage ?? 'error'}`, blocks, restBlocks: [] }
  }

  // streaming / done
  const mainBlocks: unknown[] = []
  const tool = renderToolHistory(state.toolCalls)
  if (tool) mainBlocks.push(tool)

  if (state.reasoningText && state.phase === 'streaming') {
    const tail = state.reasoningText.slice(-500)
    mainBlocks.push({
      type: 'context',
      elements: [{ type: 'mrkdwn', text: `🤔 _${escapeMd(tail)}_` }],
    })
  }

  const bodyText = state.bufferedText || (state.phase === 'streaming' ? '…' : '')
  const split = bodyText ? splitMarkdown(bodyText) : { main: [], rest: [] }
  for (const b of split.main) mainBlocks.push(b)

  if (state.phase === 'done' && state.finalTotalUsage) {
    mainBlocks.push(renderCostContext(state.finalTotalUsage))
  }

  return {
    text: bodyText || '…',
    blocks: mainBlocks,
    restBlocks: split.rest,
  }
}

function renderToolHistory(calls: ToolCallRecord[]): unknown | null {
  if (calls.length === 0) return null
  const line = calls
    .map((c) => {
      const icon = c.phase === 'done' ? '✅' : c.phase === 'error' ? '❌' : '⚙️'
      return `${icon} \`${c.name || '…'}\``
    })
    .join(' · ')
  return { type: 'context', elements: [{ type: 'mrkdwn', text: `🔧 ${line}` }] }
}

function escapeMd(s: string): string {
  return s.replace(/\n/g, ' ').slice(0, 500)
}
```

```ts
import type { WebClient } from '@slack/web-api'
import type { EventSink } from '@/im/types.ts'
import type { AgentExecutionEvent } from '@/core/events.ts'
import type { Logger } from '@/logger/logger.ts'
import { initialRenderState, reduceRenderState, type RenderState } from './renderState.ts'
import { renderSlack } from './SlackRenderer.ts'
import { createSafeRender } from './safeRender.ts'
import { THINKING_ROTATION_MS } from './thinkingRotation.ts'

export interface SlackEventSinkDeps {
  web: WebClient
  channelId: string
  threadTs: string
  placeholderTs: string
  sourceMessageTs: string
  logger: Logger
}

export function createSlackEventSink(deps: SlackEventSinkDeps): EventSink {
  const safe = createSafeRender(deps.logger)
  let state: RenderState = initialRenderState()
  let lastFlush = 0
  let rotationTimer: NodeJS.Timeout | null = null

  const flush = async (force = false): Promise<void> => {
    const now = Date.now()
    if (!force && now - lastFlush < THINKING_ROTATION_MS) return
    lastFlush = now
    const out = renderSlack(state)
    await safe('chat.update', () =>
      deps.web.chat.update({
        channel: deps.channelId,
        ts: deps.placeholderTs,
        text: out.text,
        blocks: out.blocks as never,
      }),
    )
  }

  const startRotation = (): void => {
    if (rotationTimer) return
    rotationTimer = setInterval(() => {
      if (state.phase !== 'thinking') return
      state = { ...state, thinkingTick: state.thinkingTick + 1 }
      void flush(true)
    }, THINKING_ROTATION_MS)
  }

  const stopRotation = (): void => {
    if (rotationTimer) { clearInterval(rotationTimer); rotationTimer = null }
  }

  startRotation()

  return {
    emit(event: AgentExecutionEvent): void {
      state = reduceRenderState(state, event)
      if (state.phase !== 'thinking') stopRotation()
      void flush(false)
    },
    async done() {
      stopRotation()
      await flush(true)
      const out = renderSlack(state)
      for (const rest of out.restBlocks) {
        await safe('chat.postMessage rest', () =>
          deps.web.chat.postMessage({
            channel: deps.channelId,
            thread_ts: deps.threadTs,
            text: '（续）',
            blocks: rest as never,
          }),
        )
      }
      await safe('reactions.add ✅', () =>
        deps.web.reactions.add({
          channel: deps.channelId,
          timestamp: deps.sourceMessageTs,
          name: 'white_check_mark',
        }),
      )
    },
    async fail(err: Error) {
      stopRotation()
      state = reduceRenderState(state, { type: 'error', error: err })
      await flush(true)
      await safe('reactions.add ❌', () =>
        deps.web.reactions.add({
          channel: deps.channelId,
          timestamp: deps.sourceMessageTs,
```
